
# update the generic OS to the latest
sudo apt-mark hold wolfram-engine
sudo apt update
sudo apt upgrade
sudo apt dist-upgrade

# in raspi-config
enable Pi Camera
setup locale, keyboard, timezone

Encryption:
 ssh-keygen -t ed25519 -C "email@example.com"
 eval "$(ssh-agent -s)"
 ssh-add ~/.ssh/id_ed25519

sudo apt install git vim tree

git clone git@github.com:rakolina/Pikon2021.git

# in folder you will find rc.local and pishutter.py files

# add lines from rc.local into the system file
sudo vim /etc/rc.local

Place an empty file called ssh in the boot partition

Install RaspAP
curl -sL https://install.raspap.com | bash

Installing node.js

sudo apt install nodejs

> node -v
> v10.19.0

Create a service shell script called pikon-service

Run node js as a service:
chmod 755 /etc/init.d/pikon
sh /etc/init.d/pikon start/stop
update-rc.d pikon defaults

https://www.instructables.com/Nodejs-App-As-a-RPI-Service-boot-at-Startup/ 


turn off bluetooth:
  sudo systemctl stop hciuart
  sudo systemctl disable hciuart
